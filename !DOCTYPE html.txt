    <p id="questionText" style="font-size: 22px;"></p>
    <button id="btnA" onclick="selectAnswer('A')">Option A</button>
    <button id="btnB" onclick="selectAnswer('B')">Option B</button>

    <div id="explanation"></div>
    <button id="resetBtn" onclick="resetPoll()">Next Person</button>

  <script>
    const questions = [
      {
        text: "Chemical Engineering is just chemistry.",
        options: { A: "Myth", B: "Fact" },
        correct: "A",
        explanation: "Chemical engineering involves a lot more mathematics and physics, whilst also encompassing biology and material science."
      },
      {
        text: "Chemical engineers at Loughborough are all male.",
        options: { A: "Myth", B: "Fact" },
        correct: "A",
        explanation: "Chemical engineering has seen an uplift of female chemical engineers."
      },
      {
        text: "Chemical engineers mainly work in labs.",
        options: { A: "Myth", B: "Fact" },
        correct: "A",
        explanation: "Chemical engineers often work in industrial plants, design processes, and solve engineering problems â€” not just lab work."
      }
    ];

    let currentQuestion = {};

    const questionText = document.getElementById("questionText");
    const btnA = document.getElementById("btnA");
    const btnB = document.getElementById("btnB");
    const explanation = document.getElementById("explanation");
    const resetBtn = document.getElementById("resetBtn");

    function loadQuestion() {
      currentQuestion = questions[Math.floor(Math.random() * questions.length)];
      questionText.textContent = currentQuestion.text;
      btnA.textContent = currentQuestion.options.A;
      btnB.textContent = currentQuestion.options.B;
      explanation.style.display = "none";
      resetBtn.style.display = "none";
      btnA.disabled = false;
      btnB.disabled = false;
      btnA.className = "";
      btnB.className = "";
    }

    function selectAnswer(choice) {
      const correct = currentQuestion.correct;
      btnA.disabled = true;
      btnB.disabled = true;

      if (choice === correct) {
        document.getElementById(`btn${choice}`).classList.add("correct");
        document.getElementById(`btn${correct === "A" ? "B" : "A"}`).classList.add("wrong");
        explanation.innerText = "Correct! " + currentQuestion.explanation;
      } else {
        document.getElementById(`btn${choice}`).classList.add("wrong");
        document.getElementById(`btn${correct}`).classList.add("correct");
        explanation.innerText = "Incorrect. " + currentQuestion.explanation;
      }

      explanation.style.display = "block";
      resetBtn.style.display = "inline-block";
    }

    function resetPoll() {
      loadQuestion();
    }

    // Initialize the first question
    loadQuestion();
  </script>
